  <button onclick="downloadCSV()" data-nl="Download als CSV" data-en="Download as CSV">Download als CSV</button>

  <div id="output"></div>

  <script>
    let latestCode = "";
    let currentLang = "nl";

    function generateCode() {
      const isolation = document.getElementById('isolation').value;
      const subtype = document.getElementById('subtype').value;
      const location = document.getElementById('location').value;
      const side = document.getElementById('side').value;
      const notes = document.getElementById('notes').value;

      latestCode = `${isolation}.${subtype}.${location}.${side}`;

      document.getElementById('output').innerHTML =
        `<strong>PRELLIM-code:</strong> ${latestCode}<br><br>` +
        (notes ? `<em>${currentLang === 'nl' ? 'Opmerking' : 'Note'}:</em> ${notes}` : "") +
        `<br><br><small>${currentLang === 'nl' ? 'Gebruik deze code in je verslaglegging of publicatie.' : 'Use this code in your documentation or publication.'}</small>`;
    }

    function downloadCSV() {
      if (!latestCode) {
        alert(currentLang === 'nl' ? "Genereer eerst een PRELLIM-code." : "Please generate a PRELLIM code first.");
        return;
      }
      const notes = document.getElementById('notes').value;
      const csvContent = `PRELLIM-code,${currentLang === 'nl' ? 'Opmerking' : 'Note'}\n\"${latestCode}\",\"${notes}\"\n`;
      const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
      const url = URL.createObjectURL(blob);
      const link = document.createElement("a");
      link.setAttribute("href", url);
      link.setAttribute("download", "prellim_code.csv");
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
    }

    function toggleLanguage() {
      currentLang = currentLang === 'nl' ? 'en' : 'nl';
      document.querySelector('button[onclick="toggleLanguage()"]')
              .textContent = currentLang === 'nl' ? 'Switch to English' : 'Wissel naar Nederlands';

      document.querySelectorAll('[data-nl]').forEach(el => {
        el.textContent = el.getAttribute(`data-${currentLang}`);
      });

      document.getElementById('notes').placeholder =
        currentLang === 'nl'
        ? "Bijvoorbeeld: afwijking bij familiaire polydactylie"
        : "For example: anomaly in familial polydactyly";
    }
  </script>
